 1 REM THIS EXAMPLE USES STRING FUNCTIONS TO GENERATE THE DATA
 2 REM STATEMENTS FOR ANOTHER EXAMPLE.
10 A$="The quick brown fox jumps over the lazy dog."
11 C=0
15 IF C % 8 = 0 THEN PRINT "DATA ";
20 PRINT ASC(A$);
25 C=C+1
30 A$=MID$(A$, 2)
40 IF LEN(A$) = 0 THEN 60
45 IF C % 8 = 0 THEN PRINT "" ELSE PRINT ",";
50 GOTO 15
60 PRINT ""
70 PRINT "DATA 0"
